Map<String, dynamic> toJson() {
  // If many can be selected return a list
  dynamic value = multi
      ? staticOptions == null
          ? []
          : options
              .where((option) => option.isSelected)
              .map((option) => option.parameterValue)
              .toList()
      : staticOptions == null
          ? null
          : options
              .firstWhereOrNull((option) => option.isSelected)
              ?.parameterValue;

  return {parameterName: value};
}
