Future fetchCampaign(int id) async {
  try {
    var response = await http.get(domainPrefix + "campaigns/$id");
    Campaign c = Campaign.fromJson(json.decode(response.body)['data']);
    return c;
  } catch (e) {
    print(e.toString());
    return e.toString();
  }
}
